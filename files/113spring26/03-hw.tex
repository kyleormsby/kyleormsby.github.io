\begin{prob}
Let~$A$ and~$B$ be finite sets.  Explain why
\[
  |A\cup B|=|A|+|B|-|A\cap B|.
\]
\end{prob}

\begin{prob}
Let $A$, $B$, $C$ and $D$ be sets.
\begin{enumerate}[(a)]
 \item Prove that 
 \[(A \cap B) \times (C \cap D) = (A \times C) \cap (B \times D).\]
 \item Is the following equality always true?
  \[(A \cup B) \times (C \cup D) = (A \times C) \cup (B \times D)\]
  If it is, prove it. If not, provide a counterexample.
\end{enumerate}

\end{prob}
