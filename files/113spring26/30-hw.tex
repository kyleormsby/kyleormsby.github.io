\begin{prob}
Use the Fundamental Theorem of Arithmetic to prove that if $p$ is prime, $a$ and
$b$ are integers, and $p|ab$, then either $p|a$ or $p|b$ (or both).
\end{prob}

\begin{prob}
Let $p$ be a prime and let $a$ be an integer $1\le a\le p-1$.  Consider the
numbers $a,2a,3a,\ldots,(p-1)a$. Use the division algorithm to write
\[
ia =pq_i+r_i
\]
with~$0\leq r_i<p$ and integers~$q_i$ for~$1\leq i\leq p-1$.
\begin{enumerate}[(a)]
  \item Prove that~$r_i>0$ for each~$i$.
  \item If~$r_i=r_j$, show that~$p|(i-j)a$, and explain why we can then conclude
    that~$i=j$.
  \item Prove that~$\left\{ r_1,\dots,r_{p-1} \right\}=\left\{ 1,2,\dots,p-1\right\}$.
\end{enumerate}
\end{prob}
