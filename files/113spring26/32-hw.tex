\noindent
Given a positive integer $a$, denote by $\overline{a_k a_{k-1} \cdots a_1a_0}$ its digits in decimal expansion. So in particular, we know that $0\leq a_i \leq 9$ and
\[a=a_k\cdot 10^k+ a_{k-1}\cdot 10^{k-1}+ \dots + a_1\cdot 10 + a_0.\]

\begin{prob}
Prove that $a$ is divisible by 11 if and only if 
\[\sum_{i=0}^k(-1)^ia_i=a_0-a_1+\dots+(-1)^ka_k\]
 is divisible by 11.
\end{prob}

\begin{prob}
 Prove that $a$ is divisible by 7 if and only if 7 divides $\overline{a_k\dots a_1}-2a_0$ (i.e., we remove the last digit and subtract twice it from the result). 
\end{prob}
