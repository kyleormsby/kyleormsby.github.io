\begin{prob}\ 
  \begin{enumerate}[(a)]
    \item Find the smallest positive integer~$n$ such that~$7^n\equiv 1
      \pmod{100}$.

    \item Use your solution to part~(a) to find the last two digits
      of~$7^{2022}$. (You can use a computer to check your answer, but show how
	the solution can be derived easily by hand using part~(a).)

    \item (BONUS) 
      What are the last two digits of
      \[
	7^{7^{7^{\iddots^{\vspace{-5pt}7}}}}
      \]
      in which the number of~$7$s appearing is~$2022$?  Note~$7^7=823543$
      (or~$43\pmod{100}$, and~$7^{7^{7}}=7^{823543}\neq (7^7)^7=823543^7$.
  \end{enumerate}
\end{prob}

\begin{prob}
Prove that if $a,b,c,m\in \ZZ$, $c\ne 0$, and $ac\equiv bc\pmod{mc}$, then $a\equiv b\pmod{m}$.
\end{prob}
