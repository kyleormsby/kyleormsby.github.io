\begin{prob}
Consider a regular $p$-gon, and for fixed $k$ (with $1\leq k \leq p-1$), consider all $k$-subsets (subsets with $k$ elements) of all of its vertices. Define an equivalence relation on the set of $k$-subsets: two $k$-subsets are equivalent if one can be rotated into the other one.

\begin{enumerate}[(a)]
 \item Prove that if $p$ is prime, then each equivalence class has exactly $p$ elements.
 \item Show by an example that (a) is not true if $p$ is not prime.
 \item Use (a) to prove that $p$ divides $\binom{p}{k}$.
\end{enumerate}
\end{prob}
