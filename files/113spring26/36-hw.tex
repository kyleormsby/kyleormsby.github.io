\begin{prob}
Use Sunzi's Theorem to efficiently compute the congruence class of $17^2$ modulo
$35$ as follows:  First compute $17^2\pmod 7$ and $17^2\pmod 5$.  Next, solve a system of congruences that will give you the answer.\end{prob}

\begin{prob}
  Describe \emph{all} integer solutions to the system of congruences:
  \begin{align*}
    x&=1\pmod 3\\
    x&=2\pmod 4\\
    x&=3\pmod 5.
  \end{align*}
\end{prob}

\begin{prob}
  Find integers $x,y\in \{0,1,2,\ldots,7\}$ satisfying
  \begin{align*}
    x+5y&=7\pmod 8\\
    3x+y&=1\pmod 8.
  \end{align*}
\end{prob}
